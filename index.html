Installation

# from NPM

npm i aptoide-scraper

# from JSR

bunx jsr add aptoide-scraper

Usage

Get MRZ

const { 
search
} = require('aptoide-scraper')

let search = await search('whatsapp')
console.log(search)

Extract and Save Photo

const { 
download
} = require('aptoide-scraper')

let data = await download('com.whatsapp')
console.log(data)
